create table for user:

CREATE TABLE `user` (
  `user_id` varchar(36) NOT NULL,
  `username` varchar(45) NOT NULL,
  `email` varchar(255) NOT NULL,
  `password` varchar(255) NOT NULL,
  `forename` varchar(255) NOT NULL,
  `surname` varchar(255) NOT NULL,
  `dob` datetime NOT NULL,
  `address` varchar(255) NOT NULL,
  `current_balance` decimal(65,2) NOT NULL,
  PRIMARY KEY (`user_id`),
  UNIQUE KEY `user_id_UNIQUE` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

create table for security_questions:

CREATE TABLE `security_questions` (
  `user_id` varchar(255) NOT NULL,
  `question` varchar(255) NOT NULL,
  `answer` varchar(255) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

create table for budget:

CREATE TABLE `budget` (
  `user_id` varchar(255) NOT NULL,
  `budget_name` varchar(255) NOT NULL,
  PRIMARY KEY (`budget_name`),
  UNIQUE KEY `budget_name_UNIQUE` (`budget_name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

create table for budget_items:

CREATE TABLE `budget_items` (
  `budget_name` varchar(255) NOT NULL,
  `item_name` varchar(255) NOT NULL,
  `cost_budget` decimal(65, 2) NOT NULL,
  `priority` decimal(65,7) NOT NULL,
  PRIMARY KEY (`item_name`),
  UNIQUE KEY `item_name_UNIQUE` (`item_name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

create table for item:

CREATE TABLE `item` (
  `item_name` varchar(255) NOT NULL,
  `description` varchar(255) DEFAULT NULL,
  `default_cost` decimal(65, 2) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

create table for monthly costs:

CREATE TABLE `monthly_costs` (
  `budget_name` varchar(255) NOT NULL,
  `month` int NOT NULL,
  `year` int NOT NULL,
  PRIMARY KEY (`month`,`year`),
  UNIQUE KEY `month_UNIQUE` (`month`),
  UNIQUE KEY `year_UNIQUE` (`year`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

create table for transactions:

CREATE TABLE `transactions` (
  `transaction_id` varchar(255) NOT NULL,
  `transaction_name` varchar(255) NOT NULL,
  `transaction_type` varchar(255) NOT NULL,
  `amount` decimal(65, 2) NOT NULL,
  `date` datetime NOT NULL,
  `month` int NOT NULL,
  `year` int NOT NULL,
  PRIMARY KEY (`transaction_id`),
  UNIQUE KEY `transaction_id_UNIQUE` (`transaction_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
